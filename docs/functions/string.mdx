# String

Explore available string utilities.

:::info
The default string functions without types are:

```ts
/**
 * Convert a string to <case>.
 * 
 * @param input - The input string to convert.
 * @returns The converted string.
 */
export function <name>(input: string): string;
```

:::

## altCase

Convert a string to aLtErNaTiNgCaSe.

:::info
`altCase()` always starts with a lowercase letter.
:::

```js
ab.altCase('Hello.');
// => 'hElLo.'
```

---

## camelCase

Convert a string to camelCase.

```js
ab.camelCase('Hello World');
// => 'helloWorld'
```

---

## kebabCase

Convert a string to kebab-case.

```js
ab.kebabCase('Hello World');
// => 'hello-world'
```

---

## lowerCase

Convert a string to lowercase.

:::warning
The usage of `String.prototype.lowerCase()` is recommended. 
:::

```js
ab.lowerCase('HeLLo.')
// => 'hello.'
```

---

## pascalCase

Convert a string to PascalCase.

```js
ab.pascalCase('Hello World');
// => 'HelloWorld'
```

---

## sentenceCase

Convert a string to sentence case.

```js
ab.sentenceCase('hello-world');
// => 'Hello world'
```

---

## snakeCase

Convert a string to snake_case.

```js
ab.snakeCase('hello-world');
// => 'hello_world'
```

---

## startCase

Convert a string to Start Case.

```js
ab.startCase('hello world');
// => 'Hello World'
```

---

## switchCase

Switch cases in a string (lowercase to uppercase, vice versa).

```js
ab.switchCase('Hello, World!');
// => 'hELLO, wORLD!'
```

---

## titleCase

Convert a string to Title Case.

```js
ab.titleCase('fOO thE BaR.');
// => 'Foo the Bar.'
```

---

## uppercase

Convert a string to UPPERCASE.

:::warning
The usage of `String.prototype.upperCase()` is recommended. 
:::

```js
ab.upperCase('hello world');
// => 'HELLO WORLD'
```

---

## escapeHtml

Escape special HTML characters in a string.

```js
ab.escapeHtml('<div>Hello, World!</div>');
// => '&lt;div&gt;Hello, World!&lt;/div&gt;'
```

---

## unescapeHtml

Unescape escaped HTML characters in a string.

```js
ab.unescapeHtml('&lt;div&gt;Hello, World!&lt;/div&gt;');
// => '<div>Hello, World!</div>'
```

## occurences

Count the number of occurrences of a substring in a string.

```js
ab.occurences('Jerry the berry.', 'e');
// => 3
ab.occurences('Hello World!', 'Hello');
// => 1
```

```ts
/**
 * Count the number of occurrences of a substring in a string.
 * 
 * @param input - The input string to count.
 * @param substring - The substring to check in the input.
 * @returns How many times the substring was detected in the input.
 */
export function occurrences(input: string, substring: string): number;
```

---

## palindrome

Check if a string is a palindrome.

```js
ab.palindrome('hello')
// => false
ab.palindrome('radar')
// => true
```

```ts
/**
 * Check if a string is a palindrome.
 * 
 * @param input - The input string to check.
 * @returns If the input is a palindrome.
 */
export function palindrome(input: string): boolean;
```

---

## truncate

Truncate a string.

:::info
Parameter `{length}` defaults to `10` and parameter `{ending}` defaults to `'...'`.
:::

```js
ab.truncate('Hello, World!', { length: 5, ending: '...more' });
// => 'Hello...more'
ab.truncate('Hello, World!');
// => 'Hello, Wor...'
```

```ts
/**
 * Truncate a string.
 * 
 * @param input - The input string to truncate.
 * @param options - Additional options for the truncate length and ending.
 */
export function truncate(input: string, { length, ending }?: Options): string;
```
